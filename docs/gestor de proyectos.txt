-- 1. Tabla User_Role
CREATE TABLE User_Role (
    role_id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL,
    description TEXT
);

-- 2. Tabla Project_Type
CREATE TABLE Project_Type (
    type_id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL,
    description TEXT
);

-- 3. Tabla Working_Teams
CREATE TABLE Working_Teams (
    team_id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL,
    description TEXT
);

-- 4. Tabla Metodology
CREATE TABLE Metodology (
    metodology_id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100) NOT NULL,
    description TEXT
);

-- 5. Tabla User
CREATE TABLE User (
    user_id INT PRIMARY KEY AUTO_INCREMENT,
    role_id INT,
    name VARCHAR(100) NOT NULL,
    phone VARCHAR(20),
    email VARCHAR(100) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    FOREIGN KEY (role_id) REFERENCES User_Role(role_id)
);

-- 6. Tabla Project
CREATE TABLE Project (
    project_id INT PRIMARY KEY AUTO_INCREMENT,
    type_id INT,
    team_id INT,
    metodology_id INT,
    name VARCHAR(100) NOT NULL,
    description TEXT,
    budget DECIMAL(12,2),
    phase_id INT,
    FOREIGN KEY (type_id) REFERENCES Project_Type(type_id),
    FOREIGN KEY (team_id) REFERENCES Working_Teams(team_id),
    FOREIGN KEY (metodology_id) REFERENCES Metodology(metodology_id),
    FOREIGN KEY (phase_id) REFERENCES Phases(phase_id)
);

-- 7 Tabla Team_members PIVOT
CREATE TABLE Team_members(
    user_id INT,
    team_id INT,
    FOREIGN KEY (user_id) REFERENCES User(user_id),
    FOREIGN KEY (team_id) REFERENCES Working_Teams(team_id)
);
-- 8 Tabla Requirements
CREATE TABLE Requirements(
    requirement_id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100),
    description TEXT
);

-- 9 Tabla Inventory PIVOT
CREATE TABLE Inventory (
    team_id INT,
    requirement_id INT,
    FOREIGN KEY (team_id) REFERENCES Working_Teams(team_id),
    FOREIGN KEY (requirement_id) REFERENCES Requirements(requirements_id)
);
-- 10 Tabla Phases
CREATE TABLE Phases (
    phase_id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(100),
    description TEXT,
    metodology_id INT,
    FOREIGN KEY (metodology_id) REFERENCES Metodology(metodology_id)
);
